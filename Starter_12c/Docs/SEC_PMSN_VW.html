CREATE OR REPLACE VIEW SEC_PMSN_VW AS
SELECT app_cd,
       pmsn_id,
       pmsn_nm,
       pmsn_descr
  FROM sec_pmsn sp,
       app_vw   a
 WHERE sp.app_id = a.app_id
