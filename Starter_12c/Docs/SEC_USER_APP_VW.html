CREATE OR REPLACE VIEW SEC_USER_APP_VW AS
SELECT a.app_cd
      ,su.user_id
      ,su.user_nm
	  ,su.pmy_email_addr
	  ,su.work_phone
  FROM sec_user_app sua
      ,sec_user     su
      ,app          a
 WHERE sua.app_id = a.app_id
   AND sua.user_id = su.user_id
